	import torch
	import clip
	from PIL import Image
	import requests

	# 1. 设置设备 (自动检测显卡)
	device = "cuda" if torch.cuda.is_available() else "cpu"
	print(f"正在使用的设备: {device}")

	# 2. 加载模型
	print("正在加载 CLIP 模型 (ViT-B/32)...")
	model, preprocess = clip.load("ViT-B/32", device=device)

	# 3. 准备测试图片 (自动下载 CLIP 官方的测试图)
	url = "https://raw.githubusercontent.com/openai/CLIP/main/CLIP.png"
	print(f"正在下载测试图片: {url}")
	try:
    		image = Image.open(requests.get(url, stream=True).raw)
	except Exception as e:
    		print("下载图片失败，请检查网络或手动准备一张名为 'test.jpg' 的图片")
    	exit()

	# 4. 预处理图片和文本
	image_input = preprocess(image).unsqueeze(0).to(device)
	text_inputs = clip.tokenize(["a diagram", "a dog", "a cat"]).to(device)

	# 5. 推理 (计算相似度)
	print("正在进行推理...")
	with torch.no_grad():
    		image_features = model.encode_image(image_input)
    		text_features = model.encode_text(text_inputs)
    
   	 logits_per_image, logits_per_text = model(image_input, text_inputs)
    	probs = logits_per_image.softmax(dim=-1).cpu().numpy()

	# 6. 输出结果
	print("\n--- 结果 ---")
	print("标签选项: ['a diagram', 'a dog', 'a cat']")
	print(f"预测概率: {probs}")
